/* @pjs preload="img/running_man.png"; */
//PImage photo;
void setup() {
	size(600, 600); // 実行ウィンドウのサイズ
  	photo = loadImage( "img/running_man.png" );
}
void draw() {
	for(int i=0 ; i < 400 ; i++) {
    fill(128, random(256), 255); // 塗りつぶしの色をランダムに決める	// fill(赤か色相の値, 緑か彩度の値, 青か明るさの値);
    float size = random(10, 50); // 描く円のサイズを10～20の範囲で決める
    rect(random(width), random(height), size, size); // ランダムな位置に円を描画
  noLoop();
}
  image( photo, random(600), random(600), 100, 100 );
}
